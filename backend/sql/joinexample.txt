SELECT COURSE_DATA.id, COURSE_DATA.title, COURSE_DATA.coursecode, COURSE_DATA.department, COURSE_DATA.fees, COURSE_OFFER.academicyear, COURSE_OFFER.session, COURSE_OFFER.professor, COURSE_OFFER.seats
FROM COURSE_OFFER
INNER JOIN COURSE_DATA ON COURSE_OFFER.id = COURSE_DATA.courseid
WHERE COURSE_DATA.session IN ("1", "2", "3", "4", "Adjunct Fall");
